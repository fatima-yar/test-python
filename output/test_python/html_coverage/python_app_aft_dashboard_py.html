23def get_failure_list(build_list: List[jenkins_crawler.JenkinsBuildInfo]) -> List[Tuple[str, int]]: 

24 """ 

25 Create a list of all failing test cases from the given executions, sorted by the number of failures. 

26 

27 Args: 

28 build_list: List of jobs to extract failures from. 

29 

30 Returns: 

31 List of (test_case, failure_count). 

32 

33 """ 

34 # Gather and count all failing test-cases. 

35 failure_count: Dict[str, int] = {} 

36 for build in build_list: 

37 for test_case in build.aft_results: 

38 if test_case.failed(): 

39 if test_case.name not in failure_count: 

40 failure_count[test_case.name] = 0 

41 failure_count[test_case.name] += 1 

42 

43 # Sort and create a list. 

44 top_failing: List[Tuple[str, int]] = [] 

45 while len(failure_count) > 0: 

46 test_case = max(failure_count, key=failure_count.get) 

47 top_failing.append((test_case, failure_count[test_case])) 

48 failure_count.pop(test_case) 

49 

50 return top_failing

53def add_duration_plot(dashboard: Dashboard, build_info: List[jenkins_crawler.JenkinsBuildInfo]): 

54 """Add a plot to the dashboard showing the duration of each execution.""" 

55 durations: Dict[DeviceType, List[Tuple[datetime, float]]] = {} 

56 for stress_build in build_info: 

57 for variant in DeviceType.get_all(): 

58 if variant in stress_build.aft_duration: 

59 if variant not in durations: 

60 durations[variant] = [] 

61 durations[variant].append((stress_build.scheduled_at, stress_build.aft_duration[variant])) 

62 

63 # Add widget 

64 dashboard.new_row(height=600) 

65 dashboard.add_widget( 

66 DashboardLinePlot( 

67 "Duration", 

68 [DataSet(variant.name, data) for variant, data in durations.items()], 

69 "Scheduled At", 

70 "AFT Duration (minutes)", 

71 ) 

72 )

75def add_result_plot(dashboard: Dashboard, build_info: List[jenkins_crawler.JenkinsBuildInfo]): 

76 """ 

77 Add a plot to the dashboard showing the number of passes, fails etc. 

78 

79 for each date. 

80 

81 """ 

82 pass_count: List[Tuple[datetime, float]] = [] 

83 fail_count: List[Tuple[datetime, float]] = [] 

84 skip_count: List[Tuple[datetime, float]] = [] 

85 for build in build_info: 

86 pass_count.append((build.scheduled_at, len([x for x in build.aft_results if x.passed()]))) 

87 fail_count.append((build.scheduled_at, len([x for x in build.aft_results if x.failed()]))) 

88 skip_count.append((build.scheduled_at, len([x for x in build.aft_results if x.skipped()]))) 

89 

90 # Add plot. 

91 dashboard.new_row(height=600) 

92 dashboard.add_widget( 

93 DashboardLinePlot( 

94 "Results", 

95 [ 

96 DataSet("Passed", pass_count, colour="#00FF00"), 

97 DataSet("Failed", fail_count, colour="#FF0000"), 

98 DataSet("Skipped", skip_count, colour="#808080"), 

99 ], 

100 "Schedule Time", 

101 "Test Count", 

102 )

106def add_failure_table( 

107 dashboard: Dashboard, build_info: List[jenkins_crawler.JenkinsBuildInfo], test_case_limit, 

108) -> None: 

109 """ 

110 Add a table to the dashboard showing failing test cases by date. 

111 

112 Args: 

113 dashboard: Dashboard to add the table to. 

114 build_info: Jenkins build information. 

115 test_case_limit: Maximum number of failing test cases to include. 

116 If there are more failures, only the most frequent failures will be shown. 

117 

118 """ 

119 count = len(build_info) 

120 

121 columns = [build_info[x].scheduled_at for x in range(count)] 

122 columns.reverse() 

123 # Create table rows of top failing test cases. 

124 max_cell_value = 0 

125 rows: List[List[Any]] = [] 

126 for test_name, _ in get_failure_list(build_info[:count])[:test_case_limit]: 

127 row = [test_name] 

128 for date in columns: 

129 failures_on_date = 0 

130 build = next((x for x in build_info if x.scheduled_at == date), None) 

131 if build is not None: 

132 for _ in [x for x in build.aft_results if (x.name == test_name) and x.failed()]: 

133 failures_on_date += 1 

134 row.append(failures_on_date) 

135 max_cell_value = max(failures_on_date, max_cell_value) 

136 rows.append(row) 

137 

138 # Add table. 

139 dashboard.new_row() 

140 dashboard.add_widget( 

141 DashboardText( 

142 "Top {} failures over last {} executions".format(test_case_limit, count), DashboardTextFormat.HEADING_3, 

143 ) 

144 ) 

145 

146 dashboard.new_row() 

147 header_row = [""] 

148 header_row.extend(x.strftime("%b %d") for x in columns) 

149 table = DashboardTable( 

150 header_row, 

151 highlight_first_column=True, 

152 rows=rows, 

153 value_colour_scale=DashboardTableColourScale(0, max_cell_value, ["#808080", "#b00000"]), 

154 ) 

155 dashboard.add_widget(table)

158def aft_dashboard(nightly_count: int = 14, stress_count: int = 35): 

159 """ 

160 Generate an AFT dashboard, pulling historical execution results from Jenkins. 

161 

162 Args: 

163 nightly_count: Number of nightly "regression" executions to report. 

164 stress_count: Number of stress "normal" executions to report. 

165 

166 """ 

167 temporary_directory = tempfile.TemporaryDirectory() 

168 nightly_execution_info = jenkins_crawler.collect( 

169 "Hermes.Nightly/develop", maximum_count=nightly_count, temporary_directory=temporary_directory.name 

170 ) 

171 stress_execution_info = jenkins_crawler.collect( 

172 "Hermes.Stress/develop", maximum_count=stress_count, temporary_directory=temporary_directory.name 

173 ) 

174 # Produce dashboard. 

175 dashboard = Dashboard() 

176 

177 dashboard.new_row() 

178 dashboard.add_widget(DashboardText("Nightly Testing", DashboardTextFormat.HEADING_1)) 

179 dashboard.new_row() 

180 dashboard.add_widget( 

181 DashboardText("Last {} executions shown".format(len(nightly_execution_info)), DashboardTextFormat.HEADING_3) 

182 ) 

183 add_duration_plot(dashboard, nightly_execution_info) 

184 add_result_plot(dashboard, nightly_execution_info) 

185 add_failure_table(dashboard, nightly_execution_info, test_case_limit=25) 

186 

187 dashboard.new_row() 

188 dashboard.add_widget(DashboardText("\r\n\r\n\r\nStress Testing", DashboardTextFormat.HEADING_1)) 

189 dashboard.new_row() 

190 dashboard.add_widget( 

191 DashboardText("Last {} executions shown".format(len(stress_execution_info)), DashboardTextFormat.HEADING_3) 

192 ) 

193 add_duration_plot(dashboard, stress_execution_info) 

194 add_result_plot(dashboard, stress_execution_info) 

195 add_failure_table(dashboard, stress_execution_info, test_case_limit=25) 

196 

197 dashboard.publish()
